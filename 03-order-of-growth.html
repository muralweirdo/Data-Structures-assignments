def fib1(n): 
    if n <= 1: 
        return 1 
    
    else: 
        return fib1(n-2) + fib1(n-1)

def fib2(n): 
    a = 1 
    b = 1 
    
    for i in range(n): 
        a, b = b, a+b
        
    return a

import time 

def compute_times(fn, limit): 
    l = [] 
    
    for i in range(limit): 
        
        start_time = int(round(time.time() * 1000))

        fn(i)  

        end_time = int(round(time.time() * 1000))
        time_taken = end_time - start_time

        l.append(time_taken)
    
    return l
